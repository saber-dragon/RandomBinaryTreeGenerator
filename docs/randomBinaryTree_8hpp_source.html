49 TreeNode *RandomBinaryTree(size_t n, int low = 0, int high = 10000) {
50  if (n == 0) return nullptr;
51  std::vector<TreeNode *> UnsaturatedNodes(n, nullptr);
52  int Candidate = 0;
53  std::srand(std::time(nullptr)); // use current time as seed for random generator
54  int rVal = std::rand() % (high - low) + low;
55  TreeNode *root = new TreeNode(rVal);
56  --n;
57  ++Candidate;
58  UnsaturatedNodes[0] = root;
59 
60  while (n != 0) {
61  auto SIndex = std::rand() % Candidate;
62  auto Selected = UnsaturatedNodes[SIndex];
63  rVal = std::rand() % (high - low) + low;
64  auto *NewNode = new TreeNode(rVal);
65  if (!Selected->left && !Selected->right) {//
66  if (std::rand() % 2 == 0) Selected->left = NewNode;
67  else Selected->right = NewNode;
68  UnsaturatedNodes[Candidate++] = NewNode;
69  } else {
70  if (Selected->left) {
71  Selected->right = NewNode;
72  } else if (Selected->right) {
73  Selected->left = NewNode;
74  }
75  UnsaturatedNodes[SIndex] = UnsaturatedNodes[Candidate - 1];
76  UnsaturatedNodes[Candidate - 1] = NewNode;
77  }
78  --n;
79  }
80 
81  return root;
82 }
